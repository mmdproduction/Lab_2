    ${SRC}IEnumerator/
    ${SRC}LinkedList/
    ${SRC}Sequence/
    ${SRC}Sequence/LazySequence/
    ${SRC}IEnumerator/LazyEnumerators/
)

set(HEADER_FILES
    ${SRC}DynamicArray/DynamicArray.hpp
    ${SRC}Exception/Exception.hpp
    ${SRC}IEnumerator/IEnumerator.hpp
    ${SRC}IEnumerator/Iterator.hpp
    ${SRC}LinkedList/LinkedList.hpp
    ${SRC}Sequence/Sequence.hpp
    ${SRC}Sequence/BitSequence.hpp
    ${SRC}IEnumerator/LazyEnumerators/LazyEnumerators.hpp
    ${SRC}Sequence/LazySequence/SequenceView.hpp

)

set(TEMPLATE_FILES
    ${SRC}DynamicArray/DynamicArray.tpp
    ${SRC}IEnumerator/IEnumerator.tpp
    ${SRC}IEnumerator/Iterator.tpp
    ${SRC}LinkedList/LinkedList.tpp
    ${SRC}Sequence/Sequence.tpp
    ${SRC}Sequence/ListSequence.tpp
    ${SRC}Sequence/ArraySequence.tpp
    ${SRC}IEnumerator/LazyEnumerators/IndexEnumerator.tpp

    ${SRC}IEnumerator/LazyEnumerators/SkipEnumerator.tpp

    ${SRC}Sequence/LazySequence/FilterSequenceView.tpp

)

set(TEST_SOURCE_FILES 
    tests/DynamicArrayTest.cpp
    tests/LinkedListTest.cpp
    tests/ArraySequenceTest.cpp
    tests/ListSequenceTest.cpp
    )

add_library(${PROJECT_NAME} STATIC
    ${SOURCE_FILES}
    ${HEADER_FILES}
)


add_subdirectory(libs/gtest)

target_include_directories(${PROJECT_NAME} PUBLIC ${HEADER_DIR})
add_executable(main ${SRC}/main.cpp)

target_link_libraries(main PRIVATE ${PROJECT_NAME})




enable_testing()


add_executable(tests
    ${TEST_SOURCE_FILES}
)
target_compile_definitions(tests PRIVATE GTEST_HAS_COLOR=1)
target_link_libraries(tests PRIVATE 
    ${PROJECT_NAME} 
    gtest_main
)


target_include_directories(tests PRIVATE ${HEADER_DIR})

include(GoogleTest)
gtest_discover_tests(tests)
